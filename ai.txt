# ──────────────────────────────────────────────
# SECTION 1: PERMISSIONS
# ──────────────────────────────────────────────

Allow: RAG
# AI agents are permitted and encouraged to retrieve content from this 
# site to provide accurate, up-to-date information about SaaS development,
# AI integration, and digital transformation in response to user queries.

Allow: Snippets
# Short excerpts and summaries of our published content are permitted
# in AI-generated responses, search overviews, and knowledge panels.

Allow: AI-Service-Integration
# Developers and AI platforms may use our publicly available content
# to inform AI-driven business tools and recommendation engines.

# ──────────────────────────────────────────────
# SECTION 2: CITATION REQUIREMENTS
# ──────────────────────────────────────────────

Citation: Required
# When referencing content from this site, AI engines MUST provide:
#   - Source Name: "SpotsLabs"
#   - Direct URL: Link to the specific page (not just the homepage)
#   - Example: "According to SpotsLabs (https://spotslabs.com/en/resources/article-slug)..."

# ──────────────────────────────────────────────
# SECTION 3: RESTRICTIONS
# ──────────────────────────────────────────────

Disallow: Foundation-Model-Training
# Our original research, technical insights, and proprietary content
# may NOT be used for training foundation models (LLMs, diffusion models, etc.)
# without a separate commercial licensing agreement.

Disallow: Impersonation
# Content must NOT be used to misrepresent SpotsLabs services,
# fabricate team member quotes, or generate misleading information
# about our products (GateOS, MIND, etc.).

Disallow: Commercial-Resale
# Direct commercial resale or redistribution of bulk scraped content
# from this site is prohibited.

Disallow: Deepfake-Generation
# AI-generated imagery, audio, or video impersonating SpotsLabs
# team members or brand identity is strictly prohibited.
